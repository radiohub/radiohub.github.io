%%----------------------------- Housekeeping ----------------------------%%
clc
clear

%%------------------------- Simulation parameters -----------------------%%
%--- Test signal
fo = 1*1e6 ;            % Signal frequency
SNR_dB = 30 ;           % SNR [dB]

osr = 32 ;              % Oversampling rate
N_cycles = 2^2 ;        % Number of cycles to simulate

%--- ADC
N_ADC = 4 ;             % ADC resolution

%%------------------ DO NOT CHANGE CODE BELOW THIS LINE -----------------%%
%--- Establish time scale
fs = osr*2*fo ;
N_samples = N_cycles*osr ;
t = [ 0 : 1 : N_cycles*osr - 1 ] /fs;

%--- Generate noiseless test signal
s_ideal = cos(2*pi*fo*t) ;

%--- Generate AWGN
noise = ((1/sqrt(2))/10^(SNR_dB/20)) * randn( 1, N_samples ) ;

%--- Quantize signal
sout = quantizer(s_ideal+noise+0*j*(s_ideal+noise), N_ADC, 1) ;
